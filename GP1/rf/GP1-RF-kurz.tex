\subsection*{Aufgabenstellung}

\begin{enumerate}
 \item Bestimmen Sie die Abhängigkeit der Zähigkeit $\eta$ des Ethanols von der Temperatur $T$. Dazu sind für fünf verschiedene Temperaturen zwischen Raumtemperatur und \SI{50}{\degC} je zwei Messungen durchzuführen.
 \item Geben Sie die Koeffizienten $A$ und $b$ der \emph{Andradeschen Gleichung}
 \item \emph{(Für Physikstudenten)} Geben Sie die Standardabweichungen der Andradeschen Koeffizienten an. Berechnen Sie die \emph{Reynoldssche Zahl} $Re$ für die höchste Temperatur.
 \item Bestimmen Sie aus der grafischen Darstellung des natürlichen Logarithmus der Viskosität ($\ln \eta$) über dem Reziproken der Temperatur ($1/T$) den Maximalfehler des Anstiegs ($\Delta b$). Tragen Sie dazu die Fehlerflächen der Messwerte sowie die Geraden minimalen und maximalen Anstiegs in die grafische Darstellung ein.
 \item \emph{(Für Physikstudenten)} Überprüfen Sie anhand der berechneten \emph{Reynoldsschen Zahl}\footnote {O.~Reynolds 1842-1912}, ob die Strömung laminar war.
 \item Diskutieren Sie den Einfluss der fehlerhaften Größen auf den Fehler von $b$.
\end{enumerate}


\subsection*{Hinweise zur Versuchsdurchführung}

\begin{enumerate}
 \item Die Viskosität bestimmen Sie aus den Messdaten des \emph{Höppler-Viskosimeters} nach:
 \begin{refequation}{8}
  \eta = K \cdot (\rho_k - \rho_f)\cdot g\,t
  \label{rf07}
 \end{refequation}%
 %TODO: Gibt es überhaupt einen Versuchspaltz mit diesem Gerät?
 \item Beim Viskosimeter nach \emph{Ubbelohde} gilt:
 \begin{refequation}{10}
  \eta = C\,\rho_f\,g\cdot (t - t_k) \label{rf09}
 \end{refequation}%
 \item Die \emph{Andradesche Gleichung} für die Temperaturabhängigkeit der Viskosität lautet:
 \begin{refequation}{2}
  \eta(T)=A\cdot \mathrm{e}^{\frac{b}{T}}
 \label{rf02}
 \end{refequation}%
 \item Die \emph{Reynoldssche Zahl} bestimmen Sie mit:
 \begin{refequation}{11}
  Re = \frac{\rho\cdot v\cdot l^*}{\eta}
  \label{rf10}
 \end{refequation}%
 Dabei gilt für die Kugel beim Viskosimeter nach \emph{Höppler} $l^* = 2R$, der kritische (nicht zu überschreitende) Wert ist $ Re_\mathrm{krit} \approx 1000$. Für die Anordnung beim Viskosimeter nach \emph{Ubbelohde}  gilt $l^* = d$, der kritische Werte beträgt $Re_\mathrm{krit}\approx 2200$.
\end{enumerate}
