\section*{Aufgabenstellung} 

\begin{enumerate}
 \item Bestimmung der Eigenfrequenz $\omega_0$ der ungedämpften Schwingung sowie der 
 Dämpfungsfaktoren $\delta$ für verschiedene $B$-Werte der Wirbelstrombremse.
 \item Messung der Frequenzabhängigkeit von Amplitude $\hat\varphi(\omega)$ und Phase $\alpha(\omega)$ 
 mit Hilfe des Pohlschen Rades im Bereich von $f=\num{0,2}$ bis $\SI{1,2}{Hz}$ für drei verschiedene
 Dämpfungen $\left(\delta\approx\frac{\omega_0}{4}~;~\frac{\omega_0}{8}~;~\frac{\omega_0}{20}\right)$.
 \item Vergleich a.) des Güte-Wertes und der Bandbreiten mit der theoretischen Näherung und b.) einer 
 auf den Maximal-Wert normierten experimentellen Resonanzkurve mit der berechneten (in der Vorbereitung für
 $\omega_0=4\delta$ sowie $\omega_0=\SI{3,14}{Hz}$).
\end{enumerate}
